数据结构 > 线性表 > 顺序表-顺序表基本操作

用顺序表存储结构实现线性表的基本操作：

初始化 销毁 清空 检查空表 获取长度
输入位置输出值 输入值查找位置
获取前一个 获取后一个 遍历并输出
更改 插入 删除 


//初始化，销毁，清空
typedef struct Seqlist
{   //初始化顺序表
    ElemType *pData;    //动态存储空间的基地址
    int length;         //数据元素数目
    int size;           //存储空间大小
}Seqlist;

Status InitList( Seqlist &L )
{
    L.pData = (ElemType *)malloc(LISTINITSIZE*sizeof(ElemType));    //申请存储空间
    if( L.pData ==NULL ) exit(OVERFLOW);        //存储空间申请失败
    L.size = LISTINITSIZE;                      //当前已分配的存储空间大小
    L.length = 0;                               //存储数据元素个数为0
    return OK;
}   //  InitList

Status CreateList( Seqlist &L, int i )  //通过
{   //创建一个长度为i的非空表
    for( int j=0; j<i; j++ )
        L.pData[j] = (j+1)*(j+1);   //为方便测试，令元素为平方运算
    L.length = i;
    return OK;
}

Status DestroyList( Seqlist &L )
{   //销毁顺序表
    if( L.pData != NULL )
    {
        free(L.pData);
        L.pData = NULL;
    }
    L.size = 0;
    L.length = 0;
    return OK;
}   //DetroyeList

Status ClearList( Seqlist &L ) //通过
{   //清空顺序表
    L.length  = 0;
    return OK;
}   //ClearList

//访问型操作
Status ListEmpty( Seqlist L )
{   //是否为空表
    if( L.length == 0 )
        return true;
    else
        return false;
}

int ListLength( Seqlist &L )
{
    return L.length;
}

Status GetElem( Seqlist L, int i, ElemType &e )
{   //获取第i个元素
    if( i<1 || i>L.length )     //检查合法性
        return PARA_ERROR;
    e = L.pData[i-1];
    return OK;
}   //GetElem

Status LocateElem( Seqlist L, ElemType e )
{   //元素e的位置
    int i;
    for( i=0; i<L.length; i++ )
    {
        if( L.pData[i] == e )
            return i+1;
    }
    return 0;
}   //LocateElem

Status PriorElem( Seqlist L, ElemType cur_e, ElemType &pre_e )
{   //返回前驱元素
    int i;
    if( L.pData[0] == cur_e )       //为头部元素时
        return 0;
    for( i=0; i<L.length; i++ )
    {
        if( L.pData[i] == cur_e )
            pre_e = L.pData[i-1];
    }
    return OK;
}   //PriorElem

Status NextElem( Seqlist L, ElemType cur_e, ElemType &next_e )
{   //返回后继元素
    int i;
    for( i=0; i<L.length-1; i++ )
    {
        if( L.pData[i] == cur_e )
            next_e = L.pData[i+1];
    }
    if( L.pData[ L.length-1 ] == cur_e )    //为尾部元素时
        return 0;
    return OK;
}

Status ListTraverse( Seqlist L )
{   //遍历并输出
    int i;
    for( i=0; i<L.length; i++ )
        cout<<L.pData[i]<<" ";
    return OK;
}

//加工型操作
Status SetElem( Seqlist &L, int i, ElemType &e )
{   //赋值
    if( i<0 || i>L.length )
        return(PARA_ERROR);
    L.pData[i-1] = e;
    return OK;
}

Status InsertElem( Seqlist &L, int i, ElemType e )
{   //插入
    if( i<0 || i>L.length+1 )
        return(PARA_ERROR);
    if( L.length >= L.size )
    {
        ElemType *newbase = ( ElemType* ) realloc( L.pData, ( L.size+LISTINCREMENT )*sizeof( ElemType ) );
        if( newbase == NULL )   exit(OVERFLOW);
        L.pData = newbase;
        L.size += LISTINCREMENT;
    }
    //移动
    for( int j = L.length-1; j>=i-1; j-- )
        L.pData[j+1] = L.pData[j];
    L.pData[i-1] = e;
    L.length += 1;
    return OK;
}

Status DeleteElem( Seqlist &L, int i, ElemType &e )
{   //删除
    if( i<0 || i>L.length )
        return(PARA_ERROR);
    e = L.pData[i-1];
    for( int j=i; j<L.length; j++ )
        L.pData[j-1] = L.pData[j];
    L.length -= 1;
    return OK;
}
