链表的基本操作

2020-12-30- 2021-01-19
769- 4m-
数据结构 > 线性表 > 基本操作-链表顺序表基本操作

用链表存储结构实现线性表的基本操作：

初始化 销毁 清空 检查空表 获取长度
输入位置输出值 输入值查找位置
获取前一个 获取后一个 遍历并输出
更改 插入 删除 

typedef struct LNode
{
    ElemType        data;
    struct LNode    *next;
}LNode, *LinkList;

typedef struct SListInfo
{
    LinkList head;
    LinkList tail;
    LNode    *pCurNode;
    int      length;
}SListInfo;

Status InitList( SListInfo &L )
{
    L.head = (LNode *)malloc( sizeof(LNode) );
    if( L.head == NULL )    exit(OVERFLOW);
    L.head->next = NULL;
    L.tail       = L.head;
    L.pCurNode   = L.head;
    L.length     = 0;
    return OK;
}

Status CreateList( SListInfo &L, int i )
{
    for( int j = 0; j < i; j++ )
    {
        LNode* Node = (LNode *) malloc ( sizeof(LNode) );
        if(Node == NULL)    exit(0);
        Node->data = (j+1) * (j+1);
        Node->next = NULL;

        L.tail->next = Node;
        L.tail = Node;
        L.length++;
    }
    return OK;
}

Status DestroyList( SListInfo &L )
{
    while ( L.head->next != NULL )
    {
        LinkList p;
        p = L.head->next;
        L.head->next = p->next;
        free(p);
    }
    free(L.head);
    L.head      = NULL;
    L.tail      = NULL;
    L.pCurNode  = NULL;
    L.length    = 0;
    return OK;
}

Status ClearList( SListInfo &L )
{
    while ( L.head->next != NULL )
    {
        LinkList p;
        p = L.head->next;
        L.head->next = p->next;
        free(p);
    }
    L.head->next = NULL;
    L.tail       = L.head;
    L.pCurNode   = L.head;
    L.length    = 0;
    return OK;
}
///访问型操作///
Status ListEmpty( SListInfo L )
{
    if( L.length == 0 )
        return true;
    else
        return false;
}

Status ListLength( SListInfo L )
{
    return L.length;
}

Status GetElem( SListInfo &L, int i, ElemType &e )
{
    LinkList p;
    if( i<1 || i>L.length )
        return PARA_ERROR;
    p = L.head->next;
    int j = 1;
    while( j < i )
    {
        p = p->next;
        j++;
    }
    e = p->data;
    L.pCurNode = p;
    return OK;
}

Status LocateElem( SListInfo &L, ElemType e )
{
    int count = 0;
    LinkList p = L.head->next;
    while( p != NULL )
    {
        if( p->data == e )
        {
            L.pCurNode = p;
            return count+1;
            //break;
        }
        else
        {
            count++;
            p = p->next;
        }
    }
    return 0;
}

Status PriorElem( SListInfo L, ElemType e, ElemType &pre_e )
{
    LinkList q = L.head->next;
    LinkList p = q->next;
    if( L.head->next->data == e )
        return PARA_ERROR;
    while( p != NULL )    //有待考量
    {
        if( p->data == e)
        {
            pre_e = q->data;
            return OK;
        }
        else
        {
            p = p->next;
            q = q->next;
        }
    }
    return 0;
}

Status NextElem( SListInfo L, ElemType e, ElemType &next_e )
{
    LinkList p = L.head;
    if( L.tail->data == e )
        return PARA_ERROR;
    while( p->next->next != NULL )    //有待考量
    {
        if( p->data == e )
        {
            next_e = p->next->data;
            return OK;
        }
        else
            p = p->next;
    }
    return 0;
}

Status ListTraverse( SListInfo L )
{
    LinkList p = L.head->next;
    while( p != NULL )
    {
        cout<<p->data<<" ";
        p = p->next;
    }
    cout<<endl;
    return OK;
}
///加工型操作///
Status SetElem( SListInfo &L, int i, ElemType &e )
{

    L.pCurNode = L.head;
    int j = 1;
    while( j<i )
    {
        L.pCurNode = L.pCurNode->next;
        j++;
    }
    L.pCurNode->next->data = e;

    return OK;
}

Status InsertElem( SListInfo &L, int i, ElemType e )
{
    if(i > L.length + 1)
        return PARA_ERROR;

    L.pCurNode = L.head;
    int j = 1;
    while( j<i )
    {
        L.pCurNode = L.pCurNode->next;
        j++;
    }

    //申请新node
    LNode *s;
    s = (LNode *) malloc ( sizeof(LNode) );
    if( s == NULL ) exit(OVERFLOW);
    s->data = e;

    //s接在pCurNode后
    s->next             = L.pCurNode->next;
    L.pCurNode->next    = s;

    //若当前为尾结点
    if( L.pCurNode == L.tail )
    {
        L.tail = s;
        L.tail->next = NULL;
    }

    //修改链表长度
    L.length++;
    return OK;
}

Status DeleteElem( SListInfo &L, int i, ElemType &e )
{
    if( i > L.length || i < 1 )
        return PARA_ERROR;
    L.pCurNode = L.head;
    int j = 1;
    while( j < i )
    {
        L.pCurNode = L.pCurNode->next;
        j++;
    }

    e = L.pCurNode->next->data;  //赋值
    LNode *p;
    if(L.pCurNode->next->next == NULL)  //若删除的为尾结点
    {
        L.tail = L.pCurNode;
        p = L.pCurNode->next;
        free(p);
        L.tail->next = NULL;
        return OK;
    }

    p = L.pCurNode->next;
    L.pCurNode->next = p->next;
    free(p);
    L.length--;
    return OK;
}
