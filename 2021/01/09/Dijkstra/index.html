编程实现Dijkstra算法

int MIN( int *array, int *b, int n )
{
    int i, min = INFINITY, k;
    for( i = 0; i < n; i++ )
        if( b[i] == false && array[i] < min )
        {
            min = array[i];
            k = i;
        }
    return k;
}

int MAX( int *array, int n )
{
    int i, max = array[0];
    for( i = 1; i < n; i++ )
    {
        if( array[i] > max )
            max = array[i];
    }
    return max;
}

Status Dijkstra( MGraph G, int Start )
{
    int Dist[V_MAX], Path[V_MAX], Flag[V_MAX], temp[V_MAX];
    int i, j, k, MinDist, v;

    for( i = 0; i < G.VexNum; i++ )
    {   ///初始化
        Flag[i] = false;
        Dist[i] = G.arcs[Start][i];
        if( Dist[i] < INFINITY )
            Path[i] = Start;
        else
            Path[i] = -1;
    }
    Flag[Start] = true;         //选中起始点
    Path[Start] = 0;
    Dist[Start] = 0;

    //Excel( G, Dist, Path, Flag );

    for( i = 0; i < G.VexNum - 1; i++ )
    {
        //
        k = MIN( Dist, Flag, G.VexNum );   //找到 flag 为 false 的最小值
        Flag[k] = true;                    //选中
        MinDist = Dist[k];                 //当前最短路径
        //cout<<"!"<<k<<endl;
        if( MinDist < INFINITY )
        {   //更新距离
            for( j = 0; j < G.VexNum; j++ )
            {
                if( Flag[j] == false && Dist[j] > MinDist + G.arcs[k][j] )
                {   //有更小且尚未被选中，更新
                    Dist[j] = MinDist + G.arcs[k][j];
                    Path[j] = k;    //记录路径
                }
            }
        }
        //Excel( G, Dist, Path, Flag );
    }
    //输出
    Excel( G, Dist, Path, Flag );
    for( i = 0; i < G.VexNum; i++ )
    {
        if( i != Start )
        {
            temp[0] = i;
            v = Path[i];
            for( j = 1; j < G.VexNum; j++ )
            {
                temp[j] = v;
                if( v == Start )
                    break;
                v = Path[v];
            }
            temp[j] = Start;
            cout<<"从顶点 "<<Start<<" 到顶点 "<<i<<" 的最短距离为 "<<Dist[i]<<", ";
            cout<<"最短路径为 ";
            for( k = j; k > 0; k-- )
                cout<<temp[k]<<"->";
            cout<<temp[0]<<"。"<<endl;
        }
    }

}
