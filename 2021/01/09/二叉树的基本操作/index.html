二叉树的基本操作

数据结构 > 二叉树-基本操作二叉树

编程实现书P75 ADT BinaryTree 基本操作20个，
用二叉链表结构实现；

typedef struct BiTNode
{
    ElemType   data;
    struct BiTNode  *lchild, *rchild;
}BiTNode, *BiTree;

//1.初始化、销毁、清空
Status InitBiTree( BiTree &T )
{
    T = NULL;
    return OK;
}
Status DestroyeBiTree( BiTree &T )
{   //销毁//递归
    if( T != NULL )
    {   //递归
        if( T->lchild ) //存在左子树则先删除左边
            DestroyeBiTree( T->lchild );
        if( T->rchild ) //存在右子树则先删除右边
            DestroyeBiTree( T->rchild );
        free(T);    //释放
        T = NULL;   //置空
    }
    return OK;
}
Status CreateBiTree( BiTree &T )
{   //先序建立二叉树
    ElemType Inp;
    cin >>Inp;
    if( Inp == '#' )                //空结点标志
        T = NULL;
    else
    {
        T = ( BiTNode* )malloc( sizeof( BiTNode ) );    //先申请空间
        T->data = Inp;              //赋值
        CreateBiTree( T->lchild );  //先 L
        CreateBiTree( T->rchild );  //后 R
    }
    return OK;
}
Status ClearBiTree( BiTree &T )
{   //清空
    if( T != NULL )
    {   //递归
        if( T->lchild ) //存在左子树则先删除左边
            DestroyeBiTree( T->lchild );
        if( T->rchild ) //存在右子树则先删除右边
            DestroyeBiTree( T->rchild );
        free(T);
        T = NULL;
    }
    return OK;
}
//2.访问型操作
bool BiTreeEmpty( BiTree T )
{   //查空
    return ( T == NULL );
}

int BiTreeDepth( BiTree T )
{   //T的深度,递归,自下往上
    if( T == NULL ) //为空则层数为0
        return 0;
    int left  = BiTreeDepth( T->lchild );   //记录左右子树的层数
    int right = BiTreeDepth( T->rchild );
    return ( left > right )?( left + 1 ):( right + 1 ); //返回值是在左右孩子层数更多的基础上 + 1
}
Status Root( BiTree T, ElemType &e )
{   //返回T的根
    if( BiTreeEmpty(T) )
        return ERROR;
    else
        e = T->data;
    return OK;
}
Status Value( BiTree T, BiTree cur_p, ElemType &e )
{   //返回cur_p指向的结点的值
    if( cur_p == NULL )
        return ERROR;
    else
        e = cur_p->data;
    return OK;
}
Status Parent( BiTree T, BiTree cur_p, BiTree &parent )
{   //返回双亲
    parent = NULL;  //先置为空
    if( cur_p == T || T == NULL || cur_p == NULL )
        parent = NULL;
    else
    {   //利用层次遍历
        BiTree p;
        BiTree qu[MAX_SIZE];    //定义队列，存放二叉树结点指针
        int front, rear;
        front = rear = 0;       //初始化队列
        qu[ rear ]   = T;       //根结点入队
        rear++;
        while( front != rear )        //不为空进入循环
        {
            p = qu[front];            //队头出队
            front = ( front + 1 ) % MAX_SIZE;
            if( p->lchild == cur_p || p->rchild == cur_p )
            {   //判断该结点是否为双亲结点
                parent = p;
                return OK;
            }
            ///把孩子结点拉入队列
            if( p->lchild != NULL )   //存在左孩子
            {
                qu[rear] = p->lchild; //左孩子入队
                rear = ( rear + 1 ) % MAX_SIZE;
            }
            if( p->rchild != NULL )   //若存在右孩子
            {
                qu[rear] = p->rchild; //右孩子入队
                rear = ( rear + 1 ) % MAX_SIZE;
            }
        }
    }
    return OK;
}
Status LeftChild( BiTree T, BiTree cur_p, BiTree &leftChild )
{   //返回左孩子指针
    if( T == NULL || cur_p == NULL || cur_p->lchild == NULL )
        leftChild = NULL;
    else
        leftChild = cur_p->lchild;
    return OK;
}
Status RightChild( BiTree T, BiTree cur_p, BiTree &rightChild )
{   //返回右孩子指针
    if( T == NULL || cur_p == NULL || cur_p->rchild == NULL )
        rightChild = NULL;
    else
        rightChild = cur_p->rchild;
    return OK;
}
Status LeftBrother( BiTree T, BiTree cur_p, BiTree &letfBrother )
{   //返回左兄弟
    BiTree p;                //双亲
    Parent( T, cur_p, p );   //先找双亲结点
    if( p->lchild == NULL )  //为空
    {
        cout<<"没有左兄弟"<<endl;
        return ERROR;
    }
    else if( p->lchild == cur_p )
    {
        cout<<"该结点本身就是其双亲的左孩子"<<endl;
        return ERROR;
    }
    else
    {
        letfBrother = p->lchild;
        return OK;
    }
}
Status RightBrother( BiTree T, BiTree cur_p, BiTree &rightBrother )
{   //返回右兄弟
    BiTree p;
    Parent( T, cur_p, p );
    if( p->rchild == NULL )
    {
        cout<<"没有左兄弟"<<endl;
        return ERROR;
    }
    else if( p->rchild == cur_p )
    {
        cout<<"该结点本身就是其双亲的左孩子"<<endl;
        return ERROR;
    }
    else
    {
        rightBrother = p->rchild;
        return OK;
    }
}
///递归遍历
Status PreOrderTraverse( BiTree T )
{   //先序遍历 rLR
    if( T )
    {
        cout<<T->data<<" ";
        PreOrderTraverse( T->lchild );
        PreOrderTraverse( T->rchild );
    }
}
Status InOrderTraverse( BiTree T )
{   //LrR
    if( T )
    {
        InOrderTraverse( T->lchild );
        cout<<T->data<<" ";
        InOrderTraverse( T->rchild );
    }
}
Status PostOrderTraverse( BiTree T )
{   //LRr
    if( T )
    {
        PostOrderTraverse( T->lchild );
        PostOrderTraverse( T->rchild );
        cout<<T->data<<" ";
    }
}
Status LevelOrderTraverse( BiTree T )
{   //层次遍历
    BiTree p;
    BiTree qu[MAX_SIZE];    //定义队列，存放二叉树结点指针
    int front, rear;
    front = rear = 0;       //初始化队列
    qu[ rear ]   = T;       //根结点入队
    rear++;
    while( front != rear )
    {
        p = qu[front];           //队头出队
        front = ( front + 1 ) % MAX_SIZE;
        cout<<p->data<<" ";      //访问队头结点
        ///拉孩子入队
        if( p->lchild != NULL )  //存在左孩子
        {
            qu[rear] = p->lchild;//左孩子入队
            rear = ( rear + 1 ) % MAX_SIZE;
        }
        if( p->rchild != NULL )  //若存在右孩子
        {
            qu[rear] = p->rchild;//右孩子入队
            rear = ( rear + 1 ) % MAX_SIZE;
        }
    }
}
//3.加工型操作
Status Assign( BiTree T, BiTree &cur_p, ElemType value )
{   //赋值
    if( T == NULL || cur_p == NULL )
        return ERROR;
    cur_p->data = value;
    return OK;
}
Status InsertChild( BiTree T, BiTree cur_p, int LR, BiTree c )
{   //插入树
    BiTree p;
    if( T == NULL || cur_p == NULL )
        return ERROR;
    if( LR == 0 )
    {   //左孩子
        p = cur_p->lchild;
        cur_p->lchild = c;
        c->rchild = p;
        return OK;
    }
    if( LR == 1 && cur_p->rchild == NULL )
    {   //右孩子
        p = cur_p->rchild;
        cur_p->rchild = c;
        c->rchild = p;
        return OK;
    }
    return ERROR;   //其他情况均报错
}
Status DeleteChild( BiTree T, BiTree cur_p, int LR )
{   //删除左子树或右子树
    if( T == NULL )
        return ERROR;
    BiTree p;
    if( LR == 0 && cur_p->lchild != NULL )
    {
        p = cur_p->lchild;
        cur_p->lchild = NULL;
        DestroyeBiTree(p);
    }
    else if( LR == 1 && cur_p->rchild != NULL )
    {
        p = cur_p->rchild;
        cur_p->rchild = NULL;
        DestroyeBiTree(p);      //调用销毁函数
    }
    return OK;
}
