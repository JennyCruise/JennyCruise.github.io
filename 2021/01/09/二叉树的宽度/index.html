二叉树的宽度

2021-01-09- 2021-01-19
164- 1m-
数据结构 > 二叉树-二叉树

编写函数，对二叉链表结构的二叉树，求宽度。

int BiTreeWidth( BiTree T )
{   //利用层次遍历
    BiTree p;
    BiTree qu[MAX_SIZE];    //定义队列，存放二叉树结点指针
    int front, rear;
    int l = 1, len = 1, LevWid = 1, MaxWid = 1;
    front = rear = 0;       //初始化队列
    qu[ rear ]   = T;       //根结点入队
    rear++;
    while( front != rear )
    {
        l = len;
        while( l )
        {
            p = qu[front];           //队头出队
            front = ( front + 1 ) % MAX_SIZE;
            len--;
            l--;
            ///拉孩子入队
            if( p->lchild != NULL )  //存在左孩子
            {
                qu[rear] = p->lchild;//左孩子入队
                rear = ( rear + 1 ) % MAX_SIZE;
                len++;
            }
            if( p->rchild != NULL )  //若存在右孩子
            {
                qu[rear] = p->rchild;//右孩子入队
                rear = ( rear + 1 ) % MAX_SIZE;
                len++;
            }
        }
        MaxWid = ( MaxWid > len ? MaxWid : len );
    }
    return MaxWid;
}
