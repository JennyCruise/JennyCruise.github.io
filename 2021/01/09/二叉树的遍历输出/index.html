二叉链表的遍历输出

2021-01-09- 2021-01-19
581- 3m-
数据结构 > 二叉树-遍历输出

实现二叉树的先序、中序、后序遍历，
用递归和非递归方法；
实现层次遍历。

///递归遍历///
Status PreOrderTraverse( BiTree T )
{   //先序遍历 rLR
    if( T )
    {
        cout<<T->data<<" ";
        PreOrderTraverse( T->lchild );
        PreOrderTraverse( T->rchild );
    }
}
Status InOrderTraverse( BiTree T )
{   //LrR
    if( T )
    {
        InOrderTraverse( T->lchild );
        cout<<T->data<<" ";
        InOrderTraverse( T->rchild );
    }
}
Status PostOrderTraverse( BiTree T )
{   //LRr
    if( T )
    {
        PostOrderTraverse( T->lchild );
        PostOrderTraverse( T->rchild );
        cout<<T->data<<" ";
    }
}
///非递归///
Status  NonPreOrderTraverse( BiTree T )
{   //rLR
    if( T == NULL )
        return ERROR;
    stack<BiTree> s;    //栈
    BiTree p = T;       //指向根
    while( !s.empty() || p != NULL )    //栈和指针均不为空
    {
        while( p != NULL )
        {   //不断向左探，直至左孩子为空
            cout<<p->data<<" "; //输出    //r
            s.push(p);          //入栈
            p = p->lchild;      //左探    //L
        }
        //p此时为左孩子为空的结点
        if( !s.empty() )
        {   //p指向p的右孩子
            p = s.top();
            s.pop();
            p = p->rchild;                //R
        }
    }
    cout<<endl;
    return OK;
}

Status NonInOrderTraverse( BiTree T )
{   //LrR
    if( T == NULL )
        return ERROR;
    stack<BiTree> s;
    BiTree p = T;
    while( !s.empty() || p != NULL )
    {
        while( p != NULL )
        {
            s.push(p);
            p = p->lchild;      //L
        }
        if( !s.empty() )
        {
            p = s.top();
            cout<<p->data<<" "; //r
            s.pop();
            p = p->rchild;      //L
        }
    }
    cout<<endl;
    return OK;
}

Status NonPostOrderTraverse( BiTree T )
{   //LRr
    if( T == NULL )
        return ERROR;
    stack<BiTree> s;
    BiTree pCur, pPre;
    pCur = T;
    pPre = NULL;    //上一个结点，便于确认左右孩子是否被访问
    s.push( pCur );
    while( !s.empty() )
    {
        pCur = s.top();
        if( ( pCur->lchild == NULL && pCur->rchild == NULL ) || ( pPre != NULL && ( pPre == pCur->lchild || pPre == pCur->rchild ) ) )
        //  1.左右孩子都不存在                                    2.表示左右孩子被访问过
        {   //直接访问
            cout<<pCur->data<<" ";
            s.pop();        //弹栈
            pPre = pCur;    //下一个
        }
        else
        {   //左右孩子没有被访问过
            if( pCur->rchild != NULL )  //R入栈
                s.push( pCur->rchild );
            if( pCur->lchild != NULL )  //L入栈
                s.push( pCur->lchild );
        }
    }
    cout<<endl;
    return OK;
}
///层次遍历///
Status LevelOrderTraverse( BiTree T )
{   //层次遍历
    BiTree p;
    BiTree qu[MAX_SIZE];          //定义队列，存放二叉树结点指针
    int front, rear;
    front = rear = 0;             //初始化队列
    qu[ rear ]   = T;             //根结点入队
    rear++;
    while( front != rear )
    {
        p = qu[front];            //队头出队
        front = ( front + 1 ) % MAX_SIZE;
        cout<<p->data<<" ";       //访问队头结点
        if( p->lchild != NULL )   //存在左孩子
        {
            qu[rear] = p->lchild; //左孩子入队
            rear = ( rear + 1 ) % MAX_SIZE;
        }
        if( p->rchild != NULL )   //若存在右孩子
        {
            qu[rear] = p->rchild; //右孩子入队
            rear = ( rear + 1 ) % MAX_SIZE;
        }
    }
    cout<<endl;
}
